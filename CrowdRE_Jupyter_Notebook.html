def sentiment_analysis(client,sentence):
    overall_score_fifty = 0
    overall_score_thirty = 0
    documents = [sentence]
    response = client.analyze_sentiment(documents=documents)[0]
    print("Document Sentiment: {}".format(response.sentiment))
    print("Overall scores: positive={0:.2f}; neutral={1:.2f}; negative={2:.2f} \n".format(
        response.confidence_scores.positive,
        response.confidence_scores.neutral,
        response.confidence_scores.negative,
    ))
    overall_score_fifty = (response.confidence_scores.positive * 100) + (response.confidence_scores.neutral * 50)
    overall_score_thirty = (response.confidence_scores.positive * 100) + (response.confidence_scores.neutral * 30)
    return overall_score_fifty,overall_score_thirty
